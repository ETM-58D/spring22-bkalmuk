def prepare_data(path, ax):
    df = pd.read_csv(path, sep = '\s+', header = None)
    columns = [ax + 'T' + str(i) for i in range(1, df.shape[1])]
    columns.insert(0, 'class')
    df.columns = columns
    df['class'] = df['class'].astype(int)
    return df
